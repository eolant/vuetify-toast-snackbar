Vuetify Toast Snackbar Demo

Show 'hello' Show 'error' Show toast for 10 sec 

          this.$toast.success('Hello');

          this.$toast.error('Error');

          this.$toast('This is an information.', {
            timeout: 10000,
            icon: 'info'
          });
